void circle(float radius_x, float radius_y)
{
    int i = 0;
    float angle = 0.0;

    glBegin(GL_POLYGON);

    for(i = 0; i < 100; i++)
    {
        angle = 2 * PI * i / 100;
        glVertex3f (cos(angle) * radius_x, sin(angle) * radius_y, 0);
    }

    glEnd();
}

void spe_key(int key, int x, int y)
{
    switch (key)
    {

    case GLUT_KEY_LEFT:

        if(mx>=-100 && mx!=0)
        {
            mx -=2;
        }

        glutPostRedisplay();
        break;

    case GLUT_KEY_RIGHT:

        if(mx>=-100 && mx<=90)
        {
            mx +=2;
        }
        glutPostRedisplay();
        break;

    case GLUT_KEY_DOWN:
        if (my>=-70 && my!=0)
            my -=2;
        glutPostRedisplay();
        break;

    case GLUT_KEY_UP:
        if(my<=30)
            my +=2;
        glutPostRedisplay();
        break;
    default:
        break;
    }

}



int main()
{
    glutInitDisplayMode (GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize (850, 700);
    glutInitWindowPosition (250, 20);
    glutCreateWindow ("Kawai Baymax");
    init();
    glutDisplayFunc(display);
    glutSpecialFunc(spe_key);
    glutMainLoop();
    return 0;
}
